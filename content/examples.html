Examples

Simple select


$('#example-simple').selectMania({
	width: '300px'
});

Multiple select


$('#example-multiple').selectMania({
	width: '300px', 
	search: true
});

Sizes


$('#example-sizes').selectMania({
	size: 'tiny' / 'small' / 'medium' / 'large', 
	width: '150px'
});

Themes


$('#example-themes-color').selectMania({
	themes: [
		'orange', 
		'square'
	], 
	width: '400px'
});
